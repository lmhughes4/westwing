041    public class XMLPrinter extends Printer {
042        public static final String DEFAULT_ENCODING   = "utf-8";
043        public static final String DEFAULT_DTD_PREFIX = "http://depfind.sourceforge.net/dtd";
044    
045        private static final BitFormat format = new BitFormat(16);
046    
047        private boolean top = true;
048    
049        public XMLPrinter(PrintWriter out) {
050            this(out, DEFAULT_ENCODING, DEFAULT_DTD_PREFIX);
051        }
052        
053        public XMLPrinter(PrintWriter out, String encoding, String dtdPrefix) {
054            super(out);
055            
056            appendHeader(encoding, dtdPrefix);
057        }
058    
059        private void appendHeader(String encoding, String dtdPrefix) {
060            append("<?xml version=\"1.0\" encoding=\"").append(encoding).append("\" ?>").eol();
061            eol();
062            append("<!DOCTYPE classfiles SYSTEM \"").append(dtdPrefix).append("/classfile.dtd\">").eol();
063            eol();
064        }

076        public void visitClassfile(Classfile classfile) {
077            indent().append("<classfile magic-number=\"0x").append(Integer.toHexString(classfile.getMagicNumber()).toUpperCase()).append("\" minor-version=\"").append(classfile.getMinorVersion()).append("\" major-version=\"").append(classfile.getMajorVersion()).append("\" access-flag=\"").append(format.format(classfile.getAccessFlag())).append("\">").eol();
078            raiseIndent();
079    
080            top = true;
081            classfile.getConstantPool().accept(this);
082            top = false;
083            
084            if (classfile.isPublic())     indent().append("<public/>").eol();
085            if (classfile.isFinal())      indent().append("<final/>").eol();
086            if (classfile.isSuper())      indent().append("<super/>").eol();
087            if (classfile.isInterface())  indent().append("<is-interface/>").eol();
088            if (classfile.isAbstract())   indent().append("<abstract/>").eol();
089            if (classfile.isSynthetic())  indent().append("<synthetic/>").eol();
090            if (classfile.isAnnotation()) indent().append("<is-annotation/>").eol();
091            if (classfile.isEnum())       indent().append("<enum/>").eol();
092    
093            indent();
094            append("<this-class>");
095            classfile.getRawClass().accept(this);
096            append("</this-class>").eol();
097    
098            indent();
099            append("<superclass>");
100            if (classfile.getSuperclassIndex() != 0) {
101                classfile.getRawSuperclass().accept(this);
102            }
103            append("</superclass>").eol();
104    
105            if (!classfile.getAllInterfaces().isEmpty()) {
106                indent().append("<interfaces>").eol();
107                raiseIndent();
108                for (Class_info class_info : classfile.getAllInterfaces()) {
109                    indent();
110                    append("<interface>");
111                    class_info.accept(this);
112                    append("</interface>").eol();
113                }
114                lowerIndent();
115                indent().append("</interfaces>").eol();
116            }
117            
118            if (!classfile.getAllFields().isEmpty()) {
119                indent().append("<fields>").eol();
120                raiseIndent();
121                for (Field_info field : classfile.getAllFields()) {
122                    field.accept(this);
123                }
124                lowerIndent();
125                indent().append("</fields>").eol();
126            }
127    
128            if (!classfile.getAllMethods().isEmpty()) {
129                indent().append("<methods>").eol();
130                raiseIndent();
131                for (Method_info method : classfile.getAllMethods()) {
132                    method.accept(this);
133                }
134                lowerIndent();
135                indent().append("</methods>").eol();
136            }
137    
138            if (!classfile.getAttributes().isEmpty()) {
139                indent().append("<attributes>").eol();
140                raiseIndent();
141                for (Attribute_info attribute : classfile.getAttributes()) {
142                    attribute.accept(this);
143                }
144                lowerIndent();
145                indent().append("</attributes>").eol();
146            }
147    
148            lowerIndent();
149            indent().append("</classfile>").eol();
150        }

152        public void visitConstantPool(ConstantPool constantPool) {
153            resetCount();
154    
155            indent().append("<constant-pool>").eol();
156            raiseIndent();
157    
158            for (ConstantPoolEntry entry : constantPool) {
159                if (entry != null) {
160                    entry.accept(this);
161                }
162                incrementCount();
163            }
164    
165            lowerIndent();
166            indent().append("</constant-pool>").eol();
167        }

178            } else {

181            }

184        public void visitFieldRef_info(FieldRef_info entry) {
185            Class_info       c   = entry.getRawClass();
186            NameAndType_info nat = entry.getRawNameAndType();
187    
188            if (top) {
189                top = false;
190                indent();
191                append("<field-ref-info index=\"").append(currentCount()).append("\">");
192                append("<class>");
193                c.accept(this);
194                append("</class>");
195                append("<type>");
196                nat.getRawType().accept(this);
197                append("</type>");
198                append("<name>");
199                nat.getRawName().accept(this);
200                append("</name>");
201                append("</field-ref-info>").eol();
202                top = true;
203            } else {
204                append(DescriptorHelper.getType(nat.getType()));
205                append(" ");
206                append(entry.getFullSignature());
207            }
208        }

362            raiseIndent();
363    
364            if (entry.isPublic())    indent().append("<public/>").eol();
365            if (entry.isProtected()) indent().append("<protected/>").eol();
366            if (entry.isPrivate())   indent().append("<private/>").eol();
367            if (entry.isStatic())    indent().append("<static/>").eol();
368            if (entry.isFinal())     indent().append("<final/>").eol();
369            if (entry.isVolatile())  indent().append("<volatile/>").eol();
370            if (entry.isTransient()) indent().append("<transient/>").eol();
371            if (entry.isSynthetic()) indent().append("<synthetic/>").eol();
372            if (entry.isEnum())      indent().append("<enum/>").eol();
373    
374            indent();
375            append("<name>");
376            entry.getRawName().accept(this);
377            append("</name>").eol();
378            
379            indent().append("<type>").append(entry.getType()).append("</type>").eol();
380    
381            if (!entry.getAttributes().isEmpty()) {
382                indent().append("<attributes>").eol();
383                raiseIndent();

393        public void visitMethod_info(Method_info entry) {
394            indent().append("<method-info access-flag=\"").append(format.format(entry.getAccessFlag())).append("\">").eol();
395            raiseIndent();
396    
397            if (entry.isPublic())       indent().append("<public/>").eol();
398            if (entry.isProtected())    indent().append("<protected/>").eol();
399            if (entry.isPrivate())      indent().append("<private/>").eol();
400            if (entry.isStatic())       indent().append("<static/>").eol();
401            if (entry.isFinal())        indent().append("<final/>").eol();
402            if (entry.isSynchronized()) indent().append("<synchronized/>").eol();
403            if (entry.isBridge())       indent().append("<bridge/>").eol();
404            if (entry.isVarargs())      indent().append("<varargs/>").eol();
405            if (entry.isNative())       indent().append("<native/>").eol();
406            if (entry.isAbstract())     indent().append("<abstract/>").eol();
407            if (entry.isStrict())       indent().append("<strict/>").eol();
408            if (entry.isSynthetic())    indent().append("<synthetic/>").eol();
409    
410            indent();
411            append("<name>");
412            entry.getRawName().accept(this);
413            append("</name>").eol();
414            
415            if (!entry.getName().equals("<init>") && !entry.getName().equals("<clinit>")) {
416                indent().append("<return-type>").append((entry.getReturnType() != null) ? entry.getReturnType() : "void").append("</return-type>").eol();
417            }
418            indent().append("<signature>").append(entry.getSignature()).append("</signature>").eol();
419    
420            if (!entry.getAttributes().isEmpty()) {
421                indent().append("<attributes>").eol();
422                raiseIndent();
423                super.visitMethod_info(entry);
424                lowerIndent();
425                indent().append("</attributes>").eol();
426            }
427    
428            lowerIndent();
429            indent().append("</method-info>").eol();
430        }

644        public void visitInstruction(Instruction instruction) {
645            indent();
646            append("<instruction pc=\"").append(instruction.getStart()).append("\" length=\"").append(instruction.getLength()).append("\"");
647            switch (instruction.getOpcode()) {
648                case 0x02: // iconst_m1
649                case 0x03: // iconst_0
650                case 0x04: // iconst_1
651                case 0x05: // iconst_2
652                case 0x06: // iconst_3
653                case 0x07: // iconst_4
654                case 0x08: // iconst_5
655                case 0x09: // lconst_0
656                case 0x0a: // lconst_1
657                case 0x0b: // fconst_0
658                case 0x0c: // fconst_1
659                case 0x0d: // fconst_2
660                case 0x0e: // dconst_0
661                case 0x0f: // dconst_1
662                    append(" value=\"").append(instruction.getValue()).append("\">");
663                    append(instruction);
664                    break;
665                case 0x10: // bipush
666                case 0x11: // sipush
667                    append(" value=\"").append(instruction.getValue()).append("\">");
668                    append(instruction).append(" ").append(instruction.getValue());
669                    break;
670                case 0x12: // ldc
671                case 0x13: // ldc_w
672                case 0x14: // ldc2_w
673                case 0xb2: // getstatic
674                case 0xb3: // putstatic
675                case 0xb4: // getfield
676                case 0xb5: // putfield
677                case 0xb6: // invokevirtual
678                case 0xb7: // invokespecial
679                case 0xb8: // invokestatic
680                case 0xb9: // invokeinterface
681                case 0xbb: // new
682                case 0xbd: // anewarray
683                case 0xc0: // checkcast
684                case 0xc1: // instanceof
685                case 0xc5: // multianewarray
686                    append(" index=\"").append(instruction.getIndex()).append("\">");
687                    append(instruction);
688                    append(" ");

783                default:
784                    append(">");
785                    append(instruction);
786                    break;
787            }
788            append("</instruction>").eol();
789        }

1009        @VisibleForTesting
1010        String escapeXMLCharacters(String text) {
1011            StringBuilder result = new StringBuilder();
1012            boolean containsControlCharacters = false;
1013    
1014            for (int i=0; i<text.length(); i++) {
1015                char c = text.charAt(i);
1016                if (c == '&') {
1017                    result.append("&amp;");
1018                } else if (c == '<') {
1019                    result.append("&lt;");
1020                } else if (c == '>') {
1021                    result.append("&gt;");
1022                } else if (Character.isISOControl(c) || c > 0x9F) {
1023                    containsControlCharacters = true;
1024                    result.append("&#x");
1025                    result.append(Integer.toString(c, 16).toUpperCase());
1026                    result.append(";");
1027                } else {
1028                    result.append(c);
1029                }
1030            }
1031    
1032            if (containsControlCharacters) {
1033                return "<![CDATA[" + result.toString() + "]]>";
1034            } else {
1035                return result.toString();
1036            }
1037        }
